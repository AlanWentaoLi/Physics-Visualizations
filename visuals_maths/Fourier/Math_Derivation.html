Exponential Form:
$$ C_n=\frac{1}{2L}\int_{-L}^{+L}f(x)e^{-in\pi x/L}dx $$ $$ f(x)=\sum_{n=-\infty}^{n=\infty} C_ne^{in\pi x/L} $$
Trigonometric Form:
$$ f(x)=\frac{a_0}{2}+\sum_{n=1}^{\infty}a_n cos(\frac{n\pi x}{L})+b_n sin(\frac{n \pi x}{L}) $$ $$ a_n=\frac{1}{L}\int_{-L}^{+L}f(x)cos(\frac{n\pi x}{L})dx $$ $$ b_n=\frac{1}{L}\int_{-L}^{+L}f(x)sin(\frac{n\pi x}{L})dx $$

Derivation of Trigonometric Fourier Series
$$ f(x)=\sum_{n=-\infty}^{+\infty} c_ne^{in\pi x/L} $$ $$ e^{-im\pi x/L} \cdot f(x)=\sum_{n=-\infty}^{+\infty} c_n \cdot e^{in\pi x/L} \cdot e^{-im\pi x/L} $$ $$ f(x) \cdot e^{-im\pi x/L} = \sum_{n=-\infty}^{+\infty} c_n \cdot e^{(i\pi x/L)(n-m)} $$ $$ \text{Integrate from -L to L} $$ $$ \int_{-L}^L f(x) \cdot e^{-im\pi x/L}dx = \sum_{n=-\infty}^{+\infty} c_n \int_{-L}^L e^{(i\pi x/L)(n-m)} dx $$ $$ \text{But due to orthonormality of complex exponentials:} $$ $$ \int_{-L}^L f(x) \cdot e^{-im\pi x/L}dx =\left\{ \begin{array}{ll} 2L \qquad \text{(if n=m)} \\ 0 \qquad \quad \text{(if n≠m)} \end{array} \right. $$ $$ \therefore \int_{-L}^L e^{(i\pi x/L)(n-m)} dx =2L\cdot\delta_{nm} $$ $$ \int_{-L}^L f(x)\cdot e^{-im\pi x/L}dx = 2L\cdot \sum_{n=-\infty}^{+\infty}c_n\cdot \delta_{mn}=2L\cdot c_m $$ $$ c_m=\frac{1}{2L}\int_{-L}^{+L}f(x)\cdot e^{-im\pi x/L}dx $$

Triangular Fourier Series Derivation
$$ f(x)=\left\{ \begin{array}{ll} \frac{-2A}{L}\cdot (x+L) \quad (-L < x < -L/2) \\ \frac{2A}{L}\cdot x \qquad \qquad (-L/2< x < L/2) \\ \frac{-2A}{L}\cdot (x-L) \quad (L/2< x < L) \end{array} \right. $$ $$ \text{f(x) is odd}$$ $$ \therefore a_n=0 $$ $$ \therefore b_n=\frac{2}{L}\int_0^L f(x)\cdot sin(\frac{n \pi x}{L})dx $$

Square Function Fourier Series Derivation
$$ f(x)=\left\{ \begin{array}{ll} -A \quad (-L < x < 0) \\ A \quad (0< x < L) \end{array} \right. $$ $$ a_n=\frac{1}{L}\int_{-L}^0 -A\cdot cos(\frac{n\pi x}{L})dx + \frac{1}{L}\int_0^L A \cdot cos(\frac{n \pi x}{L})dx $$ $$ a_n=\frac{A}{L}\frac{L}{n\pi} sin(\frac{n\pi x}{L})|_0^L=0 $$ $$ \because \text{the function is odd} $$ $$ \therefore a_n=0 $$ $$ b_n=\frac{1}{L}\int_{-L}^0 -A\cdot sin(\frac{n\pi x}{L})dx + \frac{1}{L}\int_0^L A \cdot sin(\frac{n \pi x}{L})dx $$ $$ b_n=\frac{2A}{L}\frac{L}{n\pi} cos(\frac{n\pi x}{L})|_0^L $$ $$ b_n = \frac{2A}{n\pi}[1-cos(n\pi)] $$ $$ \because cos(nx)=(-1)^n $$ $$ \therefore b_n=\frac{2A}{n\pi}(1-(-1)^n) $$ $$ \mathcal{F}(x)=\sum_{n=1}^N \frac{2A}{n\pi}(1-(-1)^n)sin(\frac{n\pi x}{L}) $$

Sawtooth Fourier Series Derivation
$$ f(x)=Ax \quad (-L < x < L) $$ $$ a_n=\frac{1}{L}\int_{-L}^{L}Ax \cdot cos(\frac{n\pi x}{L})dx $$ $$ \because a_n \text{ is odd} $$ $$ \therefore a_n=0 $$ $$ b_n=\frac{1}{L}\int_{-L}^{L} Ax \cdot sin(\frac{n\pi x}{L})dx $$ $$ \because b_n \text{ is even} $$ $$ \therefore b_n=\frac{2A}{L}\int_{0}^{L} x \cdot sin(\frac{n\pi x}{L})dx $$ $$ \text{Using integration by part gives that} $$ $$ u=x \qquad dv=sin(\frac{n\pi x}{L})dx $$ $$ du=dx \qquad v=\frac{-L}{n\pi}cos(\frac{n\pi x}{L}) $$ $$ b_n=\frac{2A}{L} [\frac{-Lx}{n\pi}cos(\frac{n\pi x}{L})-\int_0^L\frac{-L}{n\pi}cos(\frac{n\pi x}{L})dx] $$ $$ b_n=-\frac{2A}{n\pi}cos(n\pi)=-\frac{2A}{n\pi}(-1)^n $$ $$ b_n=\frac{2A}{n\pi}(-1)^{n+1} $$ $$ \mathcal{F}=\sum_{n=1}^\infty \frac{2A}{n\pi}(-1)^{n+1}sin(\frac{n\pi x}{L}) $$

Dirac-Delta Fourier Series Derivation
$$ \delta(x)=\left\{ \begin{array}{ll} +\infty \quad (x=0) \\ 0 \qquad (x \neq 0) \end{array} \right. $$ $$ b_n=\frac{1}{L}\int_{-L}^{+L}\delta(x)\cdot sin(\frac{n\pi x}{L})dx $$ $$ \because b_n \text{ is an odd function } $$ $$ \therefore b_n=0 $$ $$ a_n=\frac{1}{L}\int_{-L}^{+L} \delta (x) \cdot cos(\frac{n\pi x}{L})dx $$ $$ a_n=\frac{1}{L} $$ $$ \mathcal{F}(x)=\sum_{n=0}^\infty \frac{1}{L}cos(\frac{n\pi x}{L}) $$

Quadratic Fourier Series Derivation
$$ f(x)=Ax^2 $$ $$ b_n = \frac{1}{L}\int_{-L}^{+L} Ax^2 \cdot sin(\frac{n \pi x}{L})dx $$ $$ \because \text{the function is odd around 0} $$ $$ \therefore b_n=0 $$ $$ a_n= \frac{1}{L}\int_{-L}^{+L} Ax^2 \cdot cos(\frac{n \pi x}{L})dx $$ $$ \because \text{the function is even around 0} $$ $$ \therefore a_n=\frac{2}{L}\int_{0}^{+L} Ax^2 \cdot cos(\frac{n \pi x}{L})dx $$ $$ \text{ at } n=0 \text{ :} $$ $$ a_0 = \frac{2}{L} \int _{0}^{+L} Ax^2 dx $$ $$ a_0 = \frac{2AL^2}{3} $$ $$ \text{ at } n\neq 0 \text{ :} $$ $$ a_n = \frac{2A}{L}\int_0^L x^2\cdot cos(\frac{n\pi x}{L})dx $$ $$ u=x^2 \qquad dv=cos(\frac{n \pi x}{L})dx $$ $$ du = 2x\cdot dx \qquad v=\frac{L}{n\pi} \cdot sin(\frac{n\pi x}{L}) $$ $$ a_n = \frac{2A}{L} \cdot [\frac{Lx^2}{n\pi}\cdot sin(\frac{n\pi x}{L})|_0^L-\int_0^L\frac{2xL}{n\pi}\cdot sin(\frac{n \pi x}{L})dx] $$ $$ a_n = \frac{2A}{L} \cdot [0-\int_0^L\frac{2xL}{n\pi}\cdot sin(\frac{n \pi x}{L})dx] $$ $$ a_n = -\frac{4A}{n\pi}\int_0^L x\cdot sin(\frac{n \pi x}{L})dx $$ $$ u = x \qquad dv = sin(\frac{n\pi x}{L}) dx$$ $$ du = dx \qquad v = -\frac{L}{n\pi} \cdot cos(\frac{n\pi x}{L})dx $$ $$ a_n = \frac{4A}{n\pi} [\frac{xL}{n\pi}\cdot cos(\frac{n\pi x}{L})|_0^L + \int_0^L\frac{xL}{n\pi} \cdot cos(\frac{n\pi x}{L})dx ] $$ $$ a_n = \frac{4A}{n\pi} \frac{xL}{n\pi}\cdot cos(\frac{n\pi x}{L})|_0^L $$ $$ a_n = \frac{4AL^2}{n^2\pi^2}(-1)^n $$ $$ \mathcal{F}(x) = \sum_{n=0}^\infty \frac{4AL^2}{n^2\pi^2}(-1)^n\cdot cos(\frac{n\pi x}{L}) $$

Linear Equation Fourier Series Derivation
$$ f(x) = Ax $$ $$ a_n = \frac{1}{L}\int_{-L}^{+L} Ax \cdot cos(\frac{n \pi x}{L})dx $$ $$ \because a_n \text{ part is an odd function} $$ $$ \therefore a_n=0 $$ $$ b_n = \frac{1}{L}\int_{-L}^{+L}Ax \cdot sin (\frac{n\pi x}{L}) dx $$ $$ b_n = \frac{A}{L}\int_{-L}^{+L}x \cdot sin (\frac{n\pi x}{L}) dx $$ $$ u=x \qquad dv = sin(\frac{n\pi x}{L})dx $$ $$ du=dx \qquad v = -\frac{L}{n\pi}\cdot cos(\frac{n\pi x}{L}) $$ $$ b_n = -\frac{A}{L}[\frac{xL}{n\pi}\cdot cos(\frac{n\pi x}{L})|_{-L}^{+L} + \int_{-L}^{+L}\frac{L}{n\pi}\cdot cos(\frac{n\pi x}{L}) dx] $$ $$ b_n = \frac{2A}{L}\cdot [-\frac{xL}{n\pi}\cdot cos(\frac{n\pi x}{L})|_{0}^{+L} + \int_{0}^{+L}\frac{L}{n\pi}\cdot cos(\frac{n\pi x}{L}) dx] $$ $$ b_n = -\frac{2AL}{n\pi}cos(n\pi) $$ $$ b_n = -\frac{2AL}{n\pi}(-1)^n $$ $$ b_n = \frac{2AL}{n\pi}(-1)^{n+1} $$ $$ \mathcal{F}(x) = \sum_{n=0}^\infty \frac{2AL}{n\pi}(-1)^{n+1} \cdot sin(\frac{n\pi x}{L}) $$

Mode x Fourier Series Derivation
$$ f(x)=A|x| $$ $$ b_n = \frac{1}{L}\int_{-L}^{+L} A|x| \cdot sin \frac{n\pi x}{L} dx $$ $$ \because \text{the function is odd around 0} $$ $$ \therefore a_n=0 $$ $$ a_n = \frac{1}{L}\int_{-L}^{+L} A|x| \cdot cos(\frac{n \pi x}{L})dx $$ $$ \because \text{the function is even around 0} $$ $$ \therefore a_n = \frac{2}{L}\int_{0}^{+L} Ax \cdot cos(\frac{n \pi x}{L})dx $$ $$ \text{ at } n=0 \text{ :} $$ $$ a_0 = \frac{2A}{L}\int_0^L x dx $$ $$ a_0 = \frac{2A}{L}\frac{x^2}{2}|_0^L $$ $$ a_0 = AL $$ $$ \text{ at } n \neq 0 \text{ :} $$ $$ a_n = \frac{2A}{L}\int_{0}^{+L} x \cdot cos(\frac{n \pi x}{L})dx $$ $$ u=x \qquad dv=cos(\frac{n\pi x}{L})dx $$ $$ du=dx \qquad v=\frac{L}{n\pi}\cdot sin(\frac{n\pi x}{L}) $$ $$ a_n = \frac{2A}{L}[\frac{Lx}{n\pi}\cdot sin(\frac{n\pi x}{L})|_0^L-\int_0^{+L}\frac{L}{n\pi}\cdot sin(\frac{n\pi x}{L})dx ]$$ $$ a_n = \frac{2A}{L}[0-\int_0^{+L}\frac{L}{n\pi}\cdot sin(\frac{n\pi x}{L})dx ]$$ $$ a_n = \frac{2A}{L}\frac{L^2}{(n\pi)^2}\cdot cos(\frac{n\pi x}{L})|_0^L $$ $$ a_n = \frac{2A}{L}\frac{L^2}{(n\pi)^2}\cdot [(-1)^n-1] $$ $$ a_n = \frac{2AL}{(n\pi)^2}[(-1)^n-1] $$
